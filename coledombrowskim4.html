def initialize_policy(grid_size):
    """Initialize policy with all directions (N, S, E, W) for every state."""
    policy = {}
    for row in range(grid_size):
        for col in range(grid_size):
            policy[(row, col)] = ['N', 'S', 'E', 'W']  # Every state starts with all directions allowed
    return policy


def initialize_value_function_and_terminal_states():
    value_function = {}
    terminal_states = {}

    # Define terminal states and rewards
    terminal_states[(0, 0)] = 5
    terminal_states[(7, 7)] = 5
    
    # Initialize value function with rewards at terminal states
    for row in range(grid_height):
        for col in range(grid_width):
            value_function[(row, col)] = terminal_states.get((row, col), 0)  # Set reward for terminal states, else 0

    return value_function, terminal_states

def draw_arrows(policy, value_function, terminal_states, grid_size):
    fig, ax = plt.subplots()
    ax.set_xticks(np.arange(grid_size))
    ax.set_yticks(np.arange(grid_size))
    ax.set_xticklabels(np.arange(grid_size))
    ax.set_yticklabels(np.arange(grid_size))
    ax.invert_yaxis()

    # Customize grid appearance
    ax.set_facecolor('#d2b48c')  # Light brown background for the grid
    
    # Remove grid lines
    plt.grid(False)

    # Loop through the grid and plot arrows based on policy, skipping arrows for terminal states but keeping value labels
    for row in range(grid_size):
        for col in range(grid_size):
            if (row, col) in terminal_states:
                # Display the value of terminal states but skip the arrows
                ax.text(col, row, f'{value_function.get((row, col), 0):.2f}', ha='center', va='center', fontsize=8, color='black')
                continue

            if (row, col) in policy:
                directions = policy[(row, col)]
                for direction in directions:
                    if direction == 'N':
                        ax.arrow(col, row, 0, -0.3, head_width=0.1, head_length=0.1, fc='orange', ec='orange')
                    elif direction == 'S':
                        ax.arrow(col, row, 0, 0.3, head_width=0.1, head_length=0.1, fc='orange', ec='orange')
                    elif direction == 'E':
                        ax.arrow(col, row, 0.3, 0, head_width=0.1, head_length=0.1, fc='orange', ec='orange')
                    elif direction == 'W':
                        ax.arrow(col, row, -0.3, 0, head_width=0.1, head_length=0.1, fc='orange', ec='orange')

            # Display value function as background color and text
            ax.text(col, row, f'{value_function.get((row, col), 0):.2f}', ha='center', va='center', fontsize=8, color='black')

    # Colorbar for value function
    img = ax.imshow(np.array([[value_function.get((r, c), 0) for c in range(grid_size)] for r in range(grid_size)]), cmap='Greens', interpolation='none')
    cbar = fig.colorbar(img, ax=ax)
    cbar.set_label('Value Function')

    plt.show()

import numpy as np
import matplotlib.pyplot as plt

def value_iteration(grid_size, rewards, discount_factor=0.9, theta=1e-4):
    actions = ['n', 's', 'e', 'w']
    actions_mapping = {
        'n': (-1, 0),  
        's': (1, 0),   
        'e': (0, 1),   
        'w': (0, -1)   
    }
    
    terminal_states = list(rewards.keys())

    value_function = np.zeros((grid_size, grid_size))

    for state, reward in rewards.items():
        value_function[state] = reward

#-----------------------------------------------------------------------------------------
    # Value iteration loop
    delta = float('inf')
    while delta >= theta:
        delta = 0
        for row in range(grid_size):
            for col in range(grid_size):
                state = (row, col)
                if state in terminal_states:
                    continue

                v = value_function[state]
                action_values = []

                for action in actions:
                    movement = actions_mapping[action]
                    next_state = (max(0, min(grid_size - 1, state[0] + movement[0])),
                                  max(0, min(grid_size - 1, state[1] + movement[1])))
                    reward = rewards.get(next_state, 0)
                    action_value = reward + discount_factor * value_function[next_state]
                    action_values.append(action_value)

                new_value = round(max(action_values), 2)
                value_function[state] = new_value
                delta = max(delta, abs(v - new_value))
    
    # Policy extraction
    policy = {}
    for row in range(grid_size):
        for col in range(grid_size):
            state = (row, col)
            if state in terminal_states:
                continue

            action_values = []
            for action in actions:
                movement = actions_mapping[action]
                next_state = (max(0, min(grid_size - 1, state[0] + movement[0])),
                              max(0, min(grid_size - 1, state[1] + movement[1])))
                reward = rewards.get(next_state, 0)
                action_value = reward + discount_factor * value_function[next_state]
                action_values.append(action_value)

            best_action_index = np.argmax(action_values)
            policy[state] = actions[best_action_index]

    return value_function, policy

def initialize_top_border_value_function(grid_height=8, grid_width=8):
    value_function = {}
    terminal_states = {}

    # Rewards
    for col in range(grid_width):
        # Top row (reward 5)
        value_function[(0, col)] = 5
        terminal_states[(0, col)] = 5 

        # Bottom row (reward -1)
        value_function[(grid_height - 1, col)] = -1
        terminal_states[(grid_height - 1, col)] = -1  

    return value_function, terminal_states

def initialize_four_corners_value_function(grid_height=8, grid_width=8):
    value_function = {}
    terminal_states = {}

    # 4 corner rewards
    corners = [(0, 0), (0, grid_width - 1), (grid_height - 1, 0), (grid_height - 1, grid_width - 1)]
    for corner in corners:
        value_function[corner] = 5
        terminal_states[corner] = 5

    # Middle penalty
    middle_state = (grid_height // 2, grid_width // 2)
    value_function[middle_state] = -20
    terminal_states[middle_state] = -20  

    return value_function, terminal_states

def initialize_middle_value_function(grid_height=8, grid_width=8):
    value_function = {}
    terminal_states = []

    # Border penalties
    for col in range(grid_width):
        if col != 0 and col != grid_width - 1:
            value_function[(0, col)] = -1  
            value_function[(grid_height - 1, col)] = -1  

            terminal_states.append((0, col))
            terminal_states.append((grid_height - 1, col))

    # Middle rewards
    middle_state = (grid_height // 2, grid_width // 2)
    value_function[middle_state] = 5
    value_function[(middle_state[0], middle_state[1] - 1)] = 5  

    terminal_states.append(middle_state)
    terminal_states.append((middle_state[0], middle_state[1] - 1))

    return value_function, terminal_states

Reflection 1:¶

Which of the three demonstrations required the most iterations before policies converged? Why do you suppose that is?¶

The "rewards in the middle" scenario would likely require more iterations due to its complexity. With the reward in the center, more surrounding cells are influenced, leading to several paths pointing towards the reward. In comparison, if the rewards were on the edges or top, fewer nearby cells would be affected, resulting in quicker convergence.

Reflection 2:¶

Do your policies make sense to you visually? Why or why not?¶

Yes, they make sense. The arrows point towards where the rewards are located: towards the middle for a central reward, towards the corners when the rewards are there, and upwards when the rewards are at the top of the grid.
